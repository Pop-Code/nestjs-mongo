NestJS Mongo

nestjs-mongo

NestJS Mongo is a module that provide a little orm. Build with typescript and the nodejs mogodb driver

Install FROM NPM

npm install nestjs-mongo
# or unig yarn
yarn add nestjs-mongo

Usage

An example of nestjs module that import the nestjs-mongo

// module.ts
import { Module } from '@nestjs/common';
import { MongoModule } from 'nestjs-mongo';

@Module({
    imports: [
        MongoModule.forRootAsync({
            imports: [],
            useFactory: (config: ConfigService) => ({
                uri: config.mongoUri
            }),
            inject: [MyConfigService]
        })
    ]
})
export class MyModule {}
